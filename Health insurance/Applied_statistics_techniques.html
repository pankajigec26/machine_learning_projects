def remove_outlier(df_in, col_name):
    df_in=df_in.sort_values(by=col_name,ascending=True)
    q1 = df_in[col_name].quantile(0.25)
    q3 = df_in[col_name].quantile(0.75)
    iqr = q3-q1 #Interquartile range
    fence_low  = q1-1.5*iqr
    print 'lowerbound-->'+str(fence_low)
    fence_high = q3+1.5*iqr
    print 'higherbound-->'+str(fence_high)
    df_outlier=[x for x in df_in[col_name] if x<fence_low or x>fence_high]
    if len(df_outlier)==0:
        print 'no outlier'
    else:
        print 'there are==>' + str(len(df_outlier))+' Outliers \n'
        print  df_outlier
    sns.boxplot(df_in[col_name])
    return df_outlier

import numpy as np

import scipy.special as sp
sns.boxplot(data = df, x = 'smoker',y='expenses',hue='smoker')

#from the chart it seems that people who smoke have genrally more expenses.
#Lets prove it statistically now 
#defining our null and alternate hypothesis
#H0=People who smoke have more expenses 
#H1=People who smoke  does not have more expenses
#Lets try with z test
#lets assume alpha level is .05
#lets take random sample of 30 

sample_data=df[df['smoker']=='yes'].sample(30)
#lets find mean of the sample_data of all the smoker 
#sample_data['smoker'].mean
sample_data['expenses'].mean()
samp_mean=sample_data['expenses'].mean()
pop_mean=df['expenses'].mean()
pop_std=df['expenses'].std()
print pop_mean
print pop_std
print samp_mean
n=30
q=.5
#lets calculate z
#z = (samp_mean - pop_mean) / np.sqrt(pop_std*pop_std/n)
z = (samp_mean - pop_mean) / np.sqrt(pop_std*pop_std / n)
print z
pval = 2 * (1 - st.norm.cdf(z))
if pval< .05 :
    print 'Null hypothesis is rejected ---- >'
else:

#Chi square test 
#Is the proportion of smokers significantly different in different genders?
#lets assume that probabilty of finding smoker and non smoker is 50-50

#state null and alternate hypothesis
H0='Smokers proportion is same across genders'
H1='Smokers proportion differ across genders'
#lets tak random samle of 30 female and 50 male and collect their  smoking habit
chi_female = df[(df['sex'] == 'female')].sample(30)

chi_male = df[(df['sex'] == 'male')].sample(50)



#grouping the male and female and count smokers and non smokers in both gender
df_male=chi_male.groupby('smoker').sex.value_counts()
df_female=chi_female.groupby('smoker').sex.value_counts()

#Putting the number of smokers and non smokers in a numpy array based on gender

dice = np.array([[df_male[0],df_male[1]],[df_female[0],df_female[1]]])

#using the chi2_contigency function from stat package 

chi2_stat, p_val, dof, ex=stats.chi2_contingency(dice)
print("===Chi2 Stat===")
print(chi2_stat)
print("\n")
print("===Degrees of Freedom===")
print(dof)
print("\n")
print("===P-Value===")
print(p_val)
print("\n")
print("===Contingency Table===")
print(ex)

#Defining critical value 
if p_val<0.05:
    print 'Null hypothesis rejected-->'+H1
else:
    print 'Null Hypothesis is Accepted -->'+H0
